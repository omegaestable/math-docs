}
\worklines{7}

\problem{2}{%
Sea $G$ un grupo conmutativo y sea $C$ el conjunto de clases conjugadas de $G$, es decir, el conjunto de clases de equivalencia bajo la relacion
\[
  g_1 \sim g_2 \quad \text{si existe } h \in G \text{ tal que } g_2 = h g_1 h^{-1}.
\]
Dada una clase conjugada $c \in C$, defina
\[
  s_c := \sum_{g \in c} g
\]
como un elemento en $\CC[G]$. Demuestre que $s_c$ es un elemento central, es decir, que $s_c$ conmuta con todos los demas elementos del algebra de grupo.
}
\worklines{11}

\problem{3}{%
Utilizando la terminologia del ejercicio anterior, demuestre que
\[
  Z(\CC[G]) = \operatorname{span}_{\CC}\langle s_c : c \in C \rangle.
\]
Donde, si $A$ es un anillo, $Z(A)$ denota el centro de $A$, es decir, el conjunto de elementos del algebra que conmutan con todos los demas.
}
\worklines{10}

\problemheader{1}{Matriz de Vandermonde e interpolacion}

Considere el problema de interpolacion de Lagrange: encontrar un polinomio $p$ de grado a lo sumo $n$ cuya grafica incluya los puntos $\{(x_i,y_i)\}_{i=0}^n$. Escriba
\[
  p(x) = \sum_{k=0}^{n} a_k x^k.
\]
Entonces interpolar equivale a encontrar los coeficientes $\{a_k\}$ que satisfacen las $n+1$ ecuaciones
\[
  p(x_i) = \sum_{k=0}^{n} a_k x_i^k = y_i, \qquad i=0,\dots,n,
\]
lo cual puede escribirse matricialmente como $A\bm{x}=\bm{b}$, donde
\[
  \bm{x} = \begin{bmatrix} a_0 \\ \vdots \\ a_n \end{bmatrix},
  \qquad
  \bm{b} = \begin{bmatrix} y_0 \\ \vdots \\ y_n \end{bmatrix},
\]
y $A$ es la matriz de Vandermonde
\[
  A = \begin{bmatrix}
  1 & x_0 & \dots & x_0^n \\
  1 & x_1 & \dots & x_1^n \\
  \vdots & \vdots & \ddots & \vdots \\
  1 & x_n & \dots & x_n^n
  \end{bmatrix} \in \R^{(n+1)\times(n+1)}.
\]
En este ejercicio se usaran nodos equidistantes en el intervalo $[-1,1]$, es decir,
\[
  x_j = -1 + 2\frac{j}{n}, \qquad j=0,\dots,n.
\]

\begin{enumerate}[label=\alph*)]
  \item Demuestre que $A$ es invertible.
  \solutionarea{5.2cm}

  \item Para $n=40$, construya la matriz $A$. Calcule el numero de condicion de $A$, su rango y su determinante. \textit{Sugerencia:} puede utilizar los comandos \texttt{vander}, \texttt{rank}, \texttt{det} y \texttt{cond} de MATLAB. Comente si la matriz es invertible desde el punto de vista numerico.
  \solutionarea{5.4cm}

  \item Para resolver el sistema $A\bm{x}=\bm{b}$, basta ejecutar el comando $\bm{\tilde{x}}=\texttt{A\textbackslash b}$. Considere $f(x)=(1+25x^2)^{-1}$ y estime
  \[
    \|f(\texttt{xx})-p(\texttt{xx})\|_\infty,
    \qquad \text{donde } \texttt{xx}=\texttt{linspace(-1,1,1e3)}.
  \]
  \solutionarea{4.9cm}

  \item Calcule la descomposicion en valores singulares de $A$ mediante
  \[
    \texttt{[U,S,V]=svd(A)}.
  \]
  De este modo,
  \[
    \tilde{x}=VS^{-1}U^Tb.
  \]
  Estime nuevamente $\|f(\texttt{xx})-p(\texttt{xx})\|_\infty$. Plotee los valores singulares de $A$ en una escala semilogaritmica, comente el comportamiento y determine el rango numerico de la matriz. \textit{Sugerencia:} la funcion \texttt{diag} puede ser de utilidad.
  \solutionarea{5.8cm}

  \item Sea $A=USV^T$ una descomposicion en valores singulares. La pseudoinversa de $A$ se define por
  \[
    A^+ = VS^+U^T,
  \]
  donde $S^+$ se obtiene invirtiendo los valores singulares positivos de $S$. Defina el vector de tolerancias
  \[
    \texttt{tol = 10.\string^(-linspace(2,17))}.
  \]
  Para cada entrada \texttt{j} de \texttt{tol}, defina $S_j$ reemplazando por cero los valores singulares menores que \texttt{tol(j)}$\sigma_1$. Calcule
  \[
    \texttt{x\_svd}=(VS_j^+U^T)b,
  \]
  y luego la norma $\|f(\texttt{xx})-p(\texttt{xx})\|_\infty$. Grafique el error en funcion del vector de tolerancias. Comente y explique el comportamiento observado.
  \solutionarea{6.5cm}

  \item Para un valor de tolerancia que minimice el error del inciso anterior, plotee en una misma grafica $f(\texttt{xx})$, $p(\texttt{xx})$ y los nodos de interpolacion. Discuta que tan buena es la aproximacion.
  \solutionarea{4.9cm}

  \item Ahora use \texttt{tol = 1e-5}. Plotee en una misma grafica $f(\texttt{xx})$, $p(\texttt{xx})$ y los nodos de interpolacion. Indique si se observa el fenomeno de Runge y explique por que.
  \solutionarea{5.0cm}
\end{enumerate}

\problemheader{2}{Compresion de imagenes mediante SVD}

En este ejercicio se utiliza la descomposicion en valores singulares para comprimir imagenes de manera basica. Una imagen de dimension $m\times n$ consiste de $mn$ pixeles. En una imagen a color, cada pixel contiene tres valores enteros $(r_p,g_p,b_p)\in[0,255]^3$, correspondientes a las componentes roja, verde y azul. Por tanto, una imagen se guarda como una matriz de tamano $m\times n\times 3$.

\begin{enumerate}[label=\alph*)]
  \item Considere la imagen suministrada \texttt{photo}. Para leerla puede usar el comando
  \[
    \texttt{RGB = imread('photo.jpg')}.
  \]
  Determine cuantos pixeles tiene la imagen almacenada en \texttt{RGB} y que tipo de entradas tiene esa matriz. \textit{Sugerencia:} ejecute \texttt{whos RGB} y comente el resultado.
  \solutionarea{4.8cm}

  \item Para utilizar la SVD, convierta la matriz \texttt{RGB} a formato doble mediante
  \[
    \texttt{RGB = im2double(RGB)}.
  \]
  \solutionarea{3.6cm}

  \item Calcule la descomposicion en valores singulares de \texttt{RGB} = $USV^T$ para cada una de las tres matrices de color. Luego calcule la aproximacion de rango $5$
  \[
    A_5 = \sum_{i=1}^{5} \sigma_i u_i v_i^T.
  \]
  En una misma ventana, usando \texttt{subplot}, muestre la imagen original, la imagen guardada en $A_5$ y el error $|\texttt{RGB}-A_5|$. Comente sus resultados. \textit{Sugerencia:} el comando \texttt{imshow(RGB)} muestra la imagen dada en \texttt{RGB}.
  \solutionarea{6.0cm}

  \item Para cada matriz \texttt{RGB(:,:,i)}, determine el numero $r_i$ de valores singulares $\sigma_k$ tales que $\sigma_k > 0.01\sigma_1$ y tome $R_1=\max r_i$. Para este valor de $R_1$, calcule la mejor aproximacion $A_{R_1}$ de rango $R_1$. Repita el procedimiento con el umbral $\sigma_k > 0.005\sigma_1$, denote el nuevo rango por $R_2$ y la aproximacion correspondiente por $A_{R_2}$. Dibuje ambas aproximaciones en una misma ventana y comente las diferencias entre ellas. \textit{Sugerencia:} el comando \texttt{diag(S)} extrae la diagonal de \texttt{S}.
  \solutionarea{6.2cm}

  \item Finalmente, guarde la imagen $A_{R_2}$ en formatos \texttt{.bmp} y \texttt{.jpg} mediante comandos del tipo
  \[
    \texttt{imwrite(A\_R2,'photo\_comp.bmp')} \quad \text{y} \quad \texttt{imwrite(A\_R2,'photo\_comp.jpg')}.
  \]
  Compare el tamano de ambos archivos con el tamano original de la imagen inicial. Recuerde que el formato \texttt{.jpg} comprime por defecto y esta optimizado para ese fin.
  \solutionarea{5.4cm}
\end{enumerate}

\problemheader{3}{Estabilidad de la factorizacion QR}

En las notas se discuten tres algoritmos para calcular la factorizacion QR de una matriz $A\in\C^{m\times n}$. Para tener una plantilla autocontenida, se reproducen los esquemas de trabajo a continuacion.

\algorithmlabel{Algoritmo 1. Ortogonalizacion de Gram-Schmidt clasica (inestable)}
\begin{tabbing}
\hspace{2.6em}\=\hspace{2.6em}\=\kill
\textbf{Entrada:} matriz $A\in\C^{m\times n}$. \\
\textbf{Salida:} matriz unitaria $Q$ y triangular superior $R$ tales que $A=QR$. \\
Para $j=1,\dots,n$: \\
\> $\bm{v}_j = \bm{a}_j$ \\
\> Para $i=1,\dots,j-1$: \\
\>\> $r_{ij} = \bm{q}_i^*\bm{a}_j$ \\
\>\> $\bm{v}_j = \bm{v}_j - r_{ij}\bm{q}_i$ \\
\> Fin para \\
\> $r_{jj} = \|\bm{v}_j\|_2$ \\
\> $\bm{q}_j = \bm{v}_j/r_{jj}$ \\
Fin para
\end{tabbing}

\algorithmlabel{Algoritmo 2. Ortogonalizacion de Gram-Schmidt modificada (estable)}
\begin{tabbing}
\hspace{2.6em}\=\hspace{2.6em}\=\kill
\textbf{Entrada:} matriz $A\in\C^{m\times n}$. \\
\textbf{Salida:} matriz unitaria $Q$ y triangular superior $R$ tales que $A=QR$. \\
$V=A$ \\
Para $i=1,\dots,n$: \\
\> $r_{ii} = \|\bm{v}_i\|_2$ \\
\> $\bm{q}_i = \bm{v}_i/r_{ii}$ \\
\> Para $j=i+1,\dots,n$: \\
\>\> $r_{ij} = \bm{q}_i^*\bm{v}_j$ \\
\>\> $\bm{v}_j = \bm{v}_j - r_{ij}\bm{q}_i$ \\
\> Fin para \\
Fin para
\end{tabbing}

\algorithmlabel{Algoritmo 3. Triangularizacion de Householder}
\begin{tabbing}
\hspace{2.6em}\=\hspace{2.6em}\=\kill
\textbf{Entrada:} matriz $A\in\C^{m\times n}$. \\
\textbf{Salida:} matriz unitaria $Q$ y triangular superior $R$ tales que $A=QR$. \\
Para $j=1,\dots,n$: \\
\> $\bm{x} = A_{j:m,j}$ \\
\> $\bm{v}_j = \operatorname{sign}(x_1)\|\bm{x}\|_2\bm{e}_1 + \bm{x}$ \\
\> $\bm{v}_j = \bm{v}_j / \|\bm{v}_j\|_2$ \\
\> $A_{j:m,j:n} = A_{j:m,j:n} - 2\bm{v}_j(\bm{v}_j^*A_{j:m,j:n})$ \\
Fin para \\
$R=A$ y $Q=I$ \\
Para $j=n,\dots,1$: \\
\> $Q_{j:m,j:n} = Q_{j:m,j:n} - 2\bm{v}_j(\bm{v}_j^*Q_{j:m,j:n})$ \\
Fin para
\end{tabbing}

\begin{enumerate}[label=\alph*)]
  \item Escriba tres funciones que implementen cada uno de los algoritmos anteriores.
  \solutionarea{4.6cm}

  \item Defina la matriz $A$ mediante las lineas
\begin{verbatim}
m = 80;
[U,~] = qr(randn(m));
[V,~] = qr(randn(m));
S = diag(2.^(-1:-1:-80));
A = U*S*V;
\end{verbatim}
  \solutionarea{3.5cm}

  \item Calcule la factorizacion $QR$ de $A$ utilizando los tres algoritmos anteriores. Para cada uno, calcule $\|A-QR\|_2$ y $\|Q^*Q-I\|_2$. Comente sus resultados, indique cual algoritmo es mejor y compare con el comando de MATLAB \texttt{[q,r] = qr(A)}.
  \solutionarea{6.6cm}
\end{enumerate}
