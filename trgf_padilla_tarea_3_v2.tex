\prob{1}%
\textit{ÂżPara cuĂˇles $n$ existe un homomorfismo sobreyectivo $S_{n+1}\to S_n$?}

\smallskip
\textbf{Respuesta.} Si y solo si $n\in\{1,2,3\}$.

Usaremos: (i) \emph{Primer teorema de isomorfismo}: si $\phi\colon G\to H$ es sobreyectivo, entonces $|\ker\phi|=|G|/|H|$.
(ii) \emph{Lema (Jordan)}: si $m\ge 5$, entonces $A_m$ es simple.

\textbf{Caso $n=1$.} $S_2\to S_1=\{e\}$: el Ăşnico homomorfismo es sobreyectivo.

\textbf{Caso $n=2$.} El signo $\operatorname{sgn}\colon S_3\to\{+1,-1\}\cong S_2$ es sobreyectivo.

\textbf{Caso $n=3$.}
Las tres particiones de $\{1,2,3,4\}$ en dos pares son
$P_1=\bigl\{\{1,2\},\{3,4\}\bigr\}$,
$P_2=\bigl\{\{1,3\},\{2,4\}\bigr\}$,
$P_3=\bigl\{\{1,4\},\{2,3\}\bigr\}$
(hay $\binom{4}{2}/2!=3$).
Cada $\sigma\in S_4$ actĂşa sobre una particiĂłn aplicando $\sigma$ a cada elemento;
como $\sigma$ es biyecciĂłn, el resultado es otra particiĂłn en dos pares,
asĂ­ que $\sigma$ permuta $\{P_1,P_2,P_3\}$.
La acciĂłn $\phi\colon S_4\to S(\{P_1,P_2,P_3\})\cong S_3$ es un homomorfismo.
AdemĂˇs, $\phi((1\;2))=(P_2\;P_3)$ y $\phi((1\;2\;3))=(P_1\;P_3\;P_2)$:
la imagen contiene una trasposiciĂłn y un $3$-ciclo, asĂ­ que genera todo $S_3$ y $\phi$ es sobreyectivo.
Por el primer teorema de isomorfismo, $|\ker\phi|=24/6=4$.
Cada doble-trasposiciĂłn fija las tres particiones, asĂ­ que $\ker\phi\supseteq V_4:=\{e,(1\;2)(3\;4),(1\;3)(2\;4),(1\;4)(2\;3)\}$; como $|V_4|=4=|\ker\phi|$, concluimos $\ker\phi=V_4$.

\textbf{Caso $n\ge 4$.}
Si existiera $\phi\colon S_{n+1}\to S_n$ sobreyectivo, $K:=\ker\phi$ serĂ­a un subgrupo normal de $S_{n+1}$ de orden $|K|=(n+1)!/n!=n+1$.
Los Ăşnicos subgrupos normales de $S_{n+1}$ para $n+1\ge 5$ son $\{e\}$, $A_{n+1}$ y $S_{n+1}$:
si $N\trianglelefteq S_{n+1}$, entonces $N\cap A_{n+1}\trianglelefteq A_{n+1}$, y por el lema de Jordan, $N\cap A_{n+1}\in\{e, A_{n+1}\}$.
Si $N\cap A_{n+1}=A_{n+1}$, entonces $N\supseteq A_{n+1}$, luego $N=A_{n+1}$ o $N=S_{n+1}$.
Si $N\cap A_{n+1}=\{e\}$, entonces para cualesquiera $\sigma,\tau\in N$ se tiene $\sigma\tau^{-1}\in N$; si ademĂˇs $\sigma\tau^{-1}\in A_{n+1}$, entonces $\sigma\tau^{-1}=e$, es decir $\sigma=\tau$. AsĂ­, cada clase lateral de $A_{n+1}$ contiene a lo mĂˇs un elemento de $N$, y por tanto $|N|\le [S_{n+1}:A_{n+1}]=2$.
Como $|K|=n+1\ge 5$, ninguno de los Ăłrdenes $1$, $(n+1)!/2$ o $(n+1)!$ es igual a $n+1$.
No existe subgrupo normal de orden $n+1$ en $S_{n+1}$: contradicciĂłn.\hfill$\square$

\prob{2}%
\textit{Sea $G$ un grupo y $C$ el conjunto de clases conjugadas. Para $c\in C$, sea $s_c:=\sum_{g\in c}g\in\CC[G]$. Demuestre que $s_c$ es central.}

\smallskip
\textbf{Respuesta.}
Basta ver que $h\,s_c\,h^{-1}=s_c$ para todo $h\in G$ (pues $G$ es base de $\CC[G]$ y la igualdad se extiende por linealidad).
Se tiene $h\,s_c\,h^{-1}=\sum_{g\in c}hgh^{-1}$.
La conjugaciĂłn por $h$ es una biyecciĂłn $c\to c$ (preserva clases conjugadas y es inyectiva sobre el conjunto finito $c$), asĂ­ que $\sum_{g\in c}hgh^{-1}=\sum_{g'\in c}g'=s_c$.
Concluimos $s_c\in Z(\CC[G])$.\hfill$\square$

\prob{3}%
\textit{Demuestre que $Z(\CC[G])=\operatorname{span}_\CC\langle s_c:c\in C\rangle$.}

\smallskip
\textbf{Respuesta.}
\textit{($\supseteq$)} Por el Problema~2, cada $s_c\in Z(\CC[G])$; como $Z(\CC[G])$ es subespacio, contiene al span.

\textit{($\subseteq$)} Sea $z=\sum_{g\in G}\alpha_g\,g\in Z(\CC[G])$.
Para todo $h\in G$: $hzh^{-1}=z$, es decir $\sum_{g}\alpha_g\,hgh^{-1}=\sum_{g}\alpha_g\,g$.
Sustituyendo $g'=hgh^{-1}$ (biyecciĂłn $G\to G$): $\sum_{g'}\alpha_{h^{-1}g'h}\,g'=\sum_{g'}\alpha_{g'}\,g'$.
Comparando coeficientes: $\alpha_{h^{-1}g'h}=\alpha_{g'}$ para todos $g',h\in G$, asĂ­ que $g\mapsto\alpha_g$ es constante en cada clase conjugada.
Denotando $\alpha_c$ el valor comĂşn para $g\in c$:
$z=\sum_{c\in C}\alpha_c\sum_{g\in c}g=\sum_{c\in C}\alpha_c\,s_c$.\hfill$\square$

\prob{4}%
\textit{Escriba $\CC[C_4]$ como suma directa de irreps.}

\smallskip
\textbf{Respuesta.}
Sea $C_4=\langle r\mid r^4=e\rangle$.
En la representaciĂłn regular, la matriz de $\rho(r)$ en la base $(e,r,r^2,r^3)$ es la permutaciĂłn cĂ­clica $P$ con $P^4=I$.
Sobre $\CC$, $x^4-1=(x-1)(x-i)(x+1)(x+i)$, asĂ­ que $P$ es diagonalizable con autovalores $\{1,i,-1,-i\}$.
Autovectores: $v_k=\bigl(1,\,(-i)^k,\,(-1)^k,\,i^k\bigr)^T$ con $Pv_k=i^k v_k$, para $k=0,1,2,3$.
Cada $\CC v_k$ es un subespacio invariante de dimensiĂłn~$1$, luego irreducible, y la representaciĂłn que induce es $\rho_k(r)=i^k$. Por tanto:
\[
  \CC[C_4]\;\cong\;(\CC,\rho_0)\oplus(\CC,\rho_1)\oplus(\CC,\rho_2)\oplus(\CC,\rho_3).
\]
ComprobaciĂłn: $\dim=1+1+1+1=4$.\hfill$\square$

\prob{5}%
\textit{Escriba $\CC[S_3]$ como suma directa de irreps.}

\smallskip
\textbf{Respuesta.}
Sea $P_3$ la representaciĂłn de permutaciĂłn sobre $\CC^3$.
Los subespacios $L:=\CC(1,1,1)$ y $V:=\{z\in\CC^3:z_1+z_2+z_3=0\}$ son $S_3$-invariantes, $L\cap V=\{0\}$, $\dim L+\dim V=3$, asĂ­ que $P_3\cong\mathbf{1}\oplus V$.

\emph{Irreducibilidad de $V$} (Steinberg, Cor.~4.3.15): $\chi_V(e)=2$, $\chi_V(\tau)=0$, $\chi_V(c)=-1$ (restando $\chi_{\mathbf{1}}\equiv 1$ de $\chi_{P_3}=\text{puntos fijos}$).
Entonces $\langle\chi_V,\chi_V\rangle=\tfrac{1}{6}(4+0+2)=1$, asĂ­ que $V$ es irreducible.

$S_3$ tiene $3$ clases conjugadas, luego exactamente $3$ irreps (Steinberg, Thm.~4.4.8).
Toda representaciĂłn $\rho\colon S_3\to\CC^\times$ factoriza por $S_3/[S_3,S_3]=S_3/A_3\cong C_2$,
dando solo $\mathbf{1}$ y $\operatorname{sgn}$.
La ecuaciĂłn $1^2+1^2+d_3^2=6$ (Steinberg, Cor.~4.4.5) da $d_3=2$: la tercera irrep es $V$.

Por la descomposiciĂłn de la representaciĂłn regular (Steinberg, Thm.~4.4.4), cada irrep aparece con multiplicidad igual a su dimensiĂłn:
\[
  \CC[S_3]\;\cong\;\mathbf{1}\;\oplus\;\operatorname{sgn}\;\oplus\; V\;\oplus\; V.
\]
ComprobaciĂłn: $1+1+2+2=6$.\hfill$\square$

\prob{6}%
\textit{Verifique que $P_4\cong\mathbf{1}\oplus V$ con $V$ irreducible de dimensiĂłn~$3$.}

\smallskip
\textbf{Respuesta.}
$L:=\CC(1,1,1,1)$ y $V:=\{z\in\CC^4:\sum z_i=0\}$ son $S_4$-invariantes, $L\cap V=\{0\}$, $\dim L+\dim V=4$, asĂ­ que $P_4\cong\mathbf{1}\oplus V$.

Las $5$ clases conjugadas de $S_4$ tienen tipos $(1^4),(2,1^2),(3,1),(4),(2^2)$ y tamaĂ±os $1,6,8,6,3$.
El carĂˇcter de $V$ es $\chi_V=\chi_{P_4}-1=(\text{puntos fijos})-1$:
\begin{gather*}
  \chi_V(e)=3,\;\;
  \chi_V((1\;2))=1,\;\;
  \chi_V((1\;2\;3))=0,\\
  \chi_V((1\;2\;3\;4))=-1,\;\;
  \chi_V((1\;2)(3\;4))=-1.
\end{gather*}
Entonces $\langle\chi_V,\chi_V\rangle=\tfrac{1}{24}(9+6+0+6+3)=1$, asĂ­ que $V$ es irreducible de dimensiĂłn~$3$ (Steinberg, Cor.~4.3.15).\hfill$\square$
